    if (bind(s, (struct sockaddr *) &server, sizeof(server)) < 0) {
        perror("Bind error:");

            continue;

            recv(c,&aux,sizeof(int),0);
            first_array[i]=aux;
            printf("%d ",aux);
        }

         int second_length, second_array[10];

        }

        printf("\nsecond array received:");
        for(int i=0;i<second_length;i++){
